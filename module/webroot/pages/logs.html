import { exec } from './kernelsu.js'; import router_state from './router.js'; const LOG_FILE = '/data/adb/modules/tcp_optimiser/service.log'; export function addLog(message) { if (!message) { return; } const timestamp = new Date().toLocaleTimeString(); const entry = `[${timestamp}] ${message}`; router_state.logsList.push(entry); if (router_state.logsList.length > 200) { router_state.logsList = router_state.logsList.slice(-200); } } function normalizeExecOutput(result) { if (typeof result === 'string') { return result; } if (!result) { return ''; } if (typeof result.stdout === 'string') { return result.stdout; } if (typeof result.out === 'string') { return result.out; } if (typeof result.output === 'string') { return result.output; } if (typeof result.result === 'string') { return result.result; } if (typeof result.stderr === 'string' && result.stderr.trim()) { return result.stderr; } try { return JSON.stringify(result, null, 2); } catch { return String(result); } } export async function read_log_file() { try { const command = ` LOG_FILE="${LOG_FILE}" if [ ! -f "$LOG_FILE" ]; then echo "service.log not found: $LOG_FILE" exit 0 fi chmod 644 "$LOG_FILE" 2>/dev/null echo "--- LOG FILE ---" ls -l "$LOG_FILE" 2>&1 echo "--- CONTENT ---" tail -n 200 "$LOG_FILE" 2>&1 `; const result = await exec(command); const output = normalizeExecOutput(result).trim(); if (output) { router_state.logsList = output.split('\n'); } else { router_state.logsList = [ 'No output returned by KernelSU exec().', 'ADB can read service.log, so this points to the WebUI exec wrapper.' ]; } return router_state.logsList; } catch (error) { console.error('Failed to read log file:', error); router_state.logsList = [ 'Failed to read service.log.', `Error: ${error && error.message ? error.message : String(error)}`, `Path: ${LOG_FILE}` ]; return router_state.logsList; } } export function updateLogsUI() { const logContent = document.getElementById('log-content'); if (!logContent) { console.warn('log-content element not found.'); return; } if (!router_state.logsList || router_state.logsList.length === 0) { logContent.textContent = 'No logs available yet.'; return; } logContent.textContent = router_state.logsList.join('\n'); logContent.scrollTop = logContent.scrollHeight; } async function clearLogs() { try { const command = ` LOG_FILE="${LOG_FILE}" mkdir -p "$(dirname "$LOG_FILE")" : > "$LOG_FILE" chmod 644 "$LOG_FILE" echo "$(date '+%Y-%m-%d %H:%M:%S') - Logs cleared from WebUI." | tee -a "$LOG_FILE" >/dev/null chmod 644 "$LOG_FILE" tail -n 20 "$LOG_FILE" 2>&1 `; const result = await exec(command); const output = normalizeExecOutput(result).trim(); if (output) { router_state.logsList = output.split('\n'); } else { await read_log_file(); } updateLogsUI(); } catch (error) { console.error('Failed to clear logs:', error); router_state.logsList = [ 'Failed to clear logs.', `Error: ${error && error.message ? error.message : String(error)}` ]; updateLogsUI(); } } function bindLogButtons() { const refreshButton = document.getElementById('refresh-logs'); const clearButton = document.getElementById('clear-logs'); if (refreshButton && refreshButton.dataset.bound !== 'true') { refreshButton.dataset.bound = 'true'; refreshButton.addEventListener('click', async () => { refreshButton.disabled = true; try { await read_log_file(); updateLogsUI(); } finally { refreshButton.disabled = false; } }); } if (clearButton && clearButton.dataset.bound !== 'true') { clearButton.dataset.bound = 'true'; clearButton.addEventListener('click', async () => { clearButton.disabled = true; try { await clearLogs(); } finally { clearButton.disabled = false; } }); } } export async function initLogs() { bindLogButtons(); await read_log_file(); updateLogsUI(); }
